void display(int a,int b);
{
    PORTB=a;
    RB0=b;
    RB1=1;
    __delay_ms(10);
    RB1 =0;
    __delay_ms(10);
    PORTB=a<<4;
    RB0=b;
    RB1=1;
    __delay_ms(10);
    RB1=0;
    __delay_ms(10);
}
